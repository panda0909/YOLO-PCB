```python
# core/detector.py
from PyQt5.QtCore import QObject, QThread, pyqtSignal
import cv2
import torch
import numpy as np

class DetectionWorker(QThread):
    frame_ready = pyqtSignal(np.ndarray)
    detection_result = pyqtSignal(dict)
    error_occurred = pyqtSignal(str)
    
    def __init__(self):
        super().__init__()
        self.model = None
        self.camera = None
        self.running = False
    
    def load_model(self, weights_path):
        try:
            # è¼‰å…¥ YOLO æ¨¡å‹
            self.model = torch.hub.load('ultralytics/yolov5', 'custom', weights_path)
            return True
        except Exception as e:
            self.error_occurred.emit(f"æ¨¡å‹è¼‰å…¥å¤±æ•—: {str(e)}")
            return False
    
    def start_camera_detection(self, camera_id=0):
        self.camera = cv2.VideoCapture(camera_id, cv2.CAP_DSHOW)
        self.running = True
        self.start()
    
    def run(self):
        while self.running and self.camera.isOpened():
            ret, frame = self.camera.read()
            if ret:
                # åŸ·è¡Œæª¢æ¸¬
                results = self.model(frame)
                
                # ç™¼é€çµæœ
                self.frame_ready.emit(frame)
                self.detection_result.emit({
                    'detections': results.pandas().xyxy[0].to_dict('records'),
                    'count': len(results.pandas().xyxy[0])
                })
            
            self.msleep(33)  # ~30 FPS
    
    def stop_detection(self):
        self.running = False
        if self.camera:
            self.camera.release()
```

```python
from PyQt5.QtCore import QObject, QThread, pyqtSignal
import torch
```

```python
class TrainingWorker(QThread):
    epoch_completed = pyqtSignal(dict)
    training_completed = pyqtSignal(str)
    error_occurred = pyqtSignal(str)
    
    def __init__(self):
        super().__init__()

    def setup_training(self, config):
        self.config = config
        # è¨­ç½®è¨“ç·´åƒæ•¸
    
    def run(self):
        # åŸ·è¡Œè¨“ç·´é‚è¼¯

                'epoch': epoch,
                'loss': 0.5 - epoch * 0.01,
                'mAP': 0.6 + epoch * 0.005
            }
            self.epoch_completed.emit(epoch_results)
            self.msleep(1000)  # æ¨¡æ“¬è¨“ç·´æ™‚é–“
        
        self.training_completed.emit("è¨“ç·´å®Œæˆ")
```
